# 1) Education vs. City
print("Method 1: Chi-square Test (Education vs City)")

if "Education" in df.columns and "City" in df.columns:
    # Create contingency table for chi-square test
    EC_table = pd.crosstab(df["Education"], df["City"])

    # Perform chi-square test
    _, p_value, _, _ = stats.chi2_contingency(EC_table)

    print(f"Chi-square test p-value = {p_value:.4e}")
    
    # Hypotheses
    print("\nNull Hypothesis (H0): Education level and City are independent.")
    print("Alternative Hypothesis (Ha): Education level and City are dependent.\n")
    
    # Conclusion
    alpha = 0.05
    if p_value < alpha:
        print(f"Since p-value ({p_value:.4e}) < alpha ({alpha}), we reject the null hypothesis.")
        print("Conclusion: There is a statistically significant association between Education and City.\n")
    else:
        print(f"Since p-value ({p_value:.4e}) >= alpha ({alpha}), we fail to reject the null hypothesis.")
        print("Conclusion: There is no statistically significant association between Education and City.\n")

    # Visualization: Stacked bar plot showing Education distribution by City
    plt.figure(figsize=(6,4))
    EC_table.plot(kind="bar", stacked=True, figsize=(6,4))
    plt.title("Education vs City")
    plt.xlabel("Education")
    plt.ylabel("Population")
    plt.legend(title="City")
    plt.show()

# 2) Education vs PaymentTier
print("\nMethod 2: ANOVA (Education vs PaymentTier)")

if "Education" in df.columns and "PaymentTier" in df.columns:
    # 1) Group data by Education and get PaymentTier values
    grouped = df.groupby("Education")["PaymentTier"]
    group_data = [grouped.get_group(x) for x in grouped.groups]

    # 2) Perform ANOVA test
    _, p_val = stats.f_oneway(*group_data)
    print(f"ANOVA test p-value = {p_val:.4e}")
    
    # Hypotheses
    print("\nNull Hypothesis (H0): The PaymentTier is the same across all Education levels.")
    print("Alternative Hypothesis (Ha): At least one Education level has a different for PaymentTier.\n")

    # Conclusion
    if p_val < alpha:
        print(f"Since p-value ({p_val:.4e}) < alpha ({alpha}), we reject the null hypothesis.")
        print("Conclusion: There is a significant difference in PaymentTier between at least two Education groups.\n")
    else:
        print(f"Since p-value ({p_val:.4e}) >= alpha ({alpha}), we fail to reject the null hypothesis.")
        print("Conclusion: There is no significant difference in PaymentTier between Education groups.\n")

    # Create contingency table for Education vs PaymentTier
    # Ensure all PaymentTier categories (1-3) are shown, even if empty
    df["PaymentTier"] = pd.Categorical(df["PaymentTier"], categories=range(1, 4), ordered=True)

    EP_table = pd.crosstab(df["Education"], df["PaymentTier"]).reindex(columns=range(1, 4), fill_value=0)
    print("Education vs PaymentTier")

    # Create stacked bar plot
    plt.figure(figsize=(6,4))
    EP_table.plot(kind="bar", stacked=True, figsize=(6,4))
    plt.title("Education vs PaymentTier")
    plt.xlabel("Education")
    plt.ylabel("Count")
    plt.legend(title="PaymentTier")
    plt.show()
